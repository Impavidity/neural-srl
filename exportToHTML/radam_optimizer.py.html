13   class RadamOptimizer(BaseOptimizer): 
14     """""" 
15      
16     #============================================================= 
17     def _init_acc(self, var_list, grads): 
18       """""" 
19        
20       super(RadamOptimizer, self)._init_acc(var_list, grads) 
21       for x_tm1, g_t in zip(var_list, grads): 
22         if self.mu > 0: 
23           self.get_accumulator(x_tm1, 'm') 
24           shape = self.get_variable_shape(x_tm1) 
25           if isinstance(g_t, tf.Tensor): 
26             self.get_accumulator(x_tm1, 'm/tm1', []) 
27           else: 
28             self.get_accumulator(x_tm1, 'm/tm1', [shape[0]]+[1]*(len(shape)-1)) 
29         if self.nu > 0: 
30           self.get_accumulator(x_tm1, 'v') 
31           shape = self.get_variable_shape(x_tm1) 
32           if isinstance(g_t, tf.Tensor): 
33             self.get_accumulator(x_tm1, 'v/tm1', []) 
34           else: 
35             self.get_accumulator(x_tm1, 'v/tm1', [shape[0]]+[1]*(len(shape)-1)) 
36       return 
37      
38     #============================================================= 
39     def _apply_dense(self, cache): 
40       """""" 
41        
42       x_tm1, g_t = cache['x_tm1'], cache['g_t'] 
43       updates = cache['updates'] 
44        
45       if self.mu > 0: 
46         m_t, t_m = self._dense_moving_average(x_tm1, g_t, 'm', beta=self.mu) 
47         m_bar_t = (1-self.gamma) * m_t + self.gamma * g_t 
48         updates.extend([m_t, t_m]) 
49       else: 
50         m_bar_t = g_t 
51        
52       if self.nu > 0: 
53         v_t, t_v = self._dense_moving_average(x_tm1, g_t**2, 'v', beta=self.nu) 
54         v_bar_t = tf.sqrt(v_t + self.epsilon) 
55         updates.extend([v_t, t_v]) 
56       else: 
57         v_bar_t = 1 
58        
59       s_t = self.learning_rate * m_bar_t / v_bar_t 
60       cache['s_t'] = s_t 
61       return cache 
62      
63     #============================================================= 
64     def _apply_sparse(self, cache): 
65       """""" 
66        
67       x_tm1, g_t, idxs = cache['x_tm1'], cache['g_t'], cache['idxs'] 
68       idxs, idxs_ = tf.unique(idxs) 
69       g_t_ = tf.unsorted_segment_sum(g_t, idxs_, tf.size(idxs)) 
70       updates = cache['updates'] 
71        
72       if self.mu > 0: 
73         m_t, t_m = self._sparse_moving_average(x_tm1, idxs, g_t_, 'm', beta=self.mu) 
74         m_t_ = tf.gather(m_t, idxs) 
75         m_bar_t_ = (1-self.gamma) * m_t_ + self.gamma * g_t_ 
76         updates.extend([m_t, t_m]) 
77       else: 
78         m_bar_t_ = g_t_ 
79        
80       if self.nu > 0: 
81         v_t, t_v = self._sparse_moving_average(x_tm1, idxs, g_t_**2, 'v', beta=self.nu) 
82         v_t_ = tf.gather(v_t, idxs) 
83         v_bar_t_ = tf.sqrt(v_t_ + self.epsilon) 
84         updates.extend([v_t, t_v]) 
85       else: 
86         v_bar_t_ = 1 
87        
88       s_t_ = self.learning_rate * m_bar_t_ / v_bar_t_ 
89       cache['s_t'] = s_t_ 
90       cache['g_t'] = g_t_ 
91       cache['idxs'] = idxs 
92       return cache
