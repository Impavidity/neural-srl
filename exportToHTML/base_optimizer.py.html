17     def __init__(self, *args, **kwargs): 
18       """""" 
19        
20       self._global_step = kwargs.pop('global_step', tf.Variable(0., trainable=False)) 
21       super(BaseOptimizer, self).__init__(*args, **kwargs) 
22       self._accumulators = {} 
23       return

26     def minimize(self, loss, name=None): 
27       """""" 
28        
29       # Error checking 
30       var_list = tf.trainable_variables() 
31       for x_tm1 in var_list: 
32         if not isinstance(x_tm1, tf.Variable): 
33           raise TypeError("Argument is not a tf.Variable: %s" % x_tm1) 
34       if not var_list: 
35         raise ValueError("No variables to optimize") 
36       if loss.dtype.base_dtype != tf.float32: 
37         raise ValueError('Loss is not float32') 
38        
39       # Compute gradients 
40       var_refs = [x_tm1._ref() for x_tm1 in var_list] 
41       grads = tf.gradients(loss, var_refs, 
42                                   colocate_gradients_with_ops=True, 
43                                   gate_gradients=True, 
44                                   aggregation_method=2) 
45       for x_tm1, g_t in zip(var_list, grads): 
46         if g_t is not None: 
47           if x_tm1.dtype.base_dtype != tf.float32: 
48             raise ValueError('%s is not float32' % x_tm1.name) 
49      
50       # Apply gradients 
51       with tf.control_dependencies(None): 
52         self._init_acc(var_list, grads) 
53       with tf.name_scope(name, self._name,[]) as name: 
54         caches = filter(lambda cache: cache['g_t'] is not None, self._prepare(var_list, grads)) 
55         for cache in caches: 
56           x_tm1, g_t = cache['x_tm1'], cache['g_t'] 
57           with tf.name_scope("update_" + x_tm1.op.name), tf.device(x_tm1.device): 
58             if isinstance(g_t, tf.Tensor): 
59               cache['g_t'] = tf.select(tf.is_finite(g_t), g_t, tf.zeros_like(g_t)) 
60               self._apply_dense(cache) 
61             else: 
62               cache['g_t'] = tf.select(tf.is_finite(g_t.values), 
63                                        g_t.values, 
64                                        tf.zeros_like(g_t.values)) 
65               cache['idxs'] = g_t.indices 
66               self._apply_sparse(cache) 
67         with tf.control_dependencies([self._finish(caches)]): 
68           with tf.device(self.global_step.device): 
69             return tf.assign_add(self.global_step, 1, name=name).op

72     def _init_acc(self, var_list, grads): 
73       """""" 
74        
75       for x_tm1, g_t in zip(var_list, grads): 
76         if self.chi > 0: 
77           tf.add_to_collection(self.get_accumulator(x_tm1, 'x'), 
78                                tf.GraphKeys.MOVING_AVERAGE_VARIABLES) 
79           shape = self.get_variable_shape(x_tm1) 
80           if isinstance(g_t, tf.Tensor): 
81             self.get_accumulator(x_tm1, 'x/tm1', []) 
82           else: 
83             self.get_accumulator(x_tm1, 'x/tm1', [shape[0]]+[1]*(len(shape)-1)) 
84       return

87     def _prepare(self, var_list, grads): 
88       """""" 
89        
90       caches = [] 
91       for x_tm1, g_t in zip(var_list, grads): 
92         caches.append({'x_tm1': x_tm1, 'g_t': g_t, 'updates': []}) 
93       return caches

108    @staticmethod 
109    def get_variable_shape(x_tm1):

113    def get_accumulator(self, x_tm1, acc_name, shape=None): 
114      """""" 
115       
116      if shape is None: 
117        shape = self.get_variable_shape(x_tm1) 
118       
119      if acc_name not in self._accumulators: 
120        self._accumulators[acc_name] = {} 
121      accumulator = self._accumulators[acc_name] 
122       
123      if x_tm1 not in accumulator: 
124        new_name = '%s/%s' % (self.name, acc_name) 
125        zeros = tf.zeros(shape, dtype=x_tm1.dtype) 
126        with tf.name_scope('%s/%s' % (x_tm1.op.name, new_name)) as scope: 
127          with tf.device(x_tm1.device): 
128            accumulator[x_tm1] = b_tm1 = tf.Variable(zeros, name=scope, trainable=False) 
129            if isinstance(x_tm1, tf.Variable) and x_tm1._save_slice_info: 
130              real_acc_name = scope[len(x_tm1.op.name + '/'):-1] 
131              slice_info = x_tm1._save_slice_info 
132              b_tm1._set_save_slice_info(tf.Variable.SaveSliceInfo( 
133                '%s/%s' % (slice_info.full_name, real_slot_name), 
134                slice_info.full_shape[:], 
135                slice_info.var_offset[:], 
136                slice_info.var_shape[:])) 
137      return accumulator[x_tm1]

140    def _dense_moving_average(self, x_tm1, a_t, name, beta=.9): 
141      """""" 
142       
143      b_tm1 = self.get_accumulator(x_tm1, '%s' % name) 
144      tm1 = self.get_accumulator(x_tm1, '%s/tm1' % name, shape=[]) 
145      t = tf.assign_add(tm1, 1) 
146      if beta < 1: 
147        beta_t = tf.convert_to_tensor(beta, name='%s/decay' % name) 
148        beta_t = beta_t * (1-beta**tm1) / (1-beta**t) 
149      else: 
150        beta_t = tm1 / t 
151      b_t = tf.assign(b_tm1, beta_t*b_tm1) 
152      b_t = tf.assign_add(b_t, (1-beta_t)*a_t) 
153      return b_t, t

156    def _sparse_moving_average(self, x_tm1, idxs, a_t_, name, beta=.9): 
157      """""" 
158       
159      b_tm1 = self.get_accumulator(x_tm1, '%s' % name) 
160      b_tm1_ = tf.gather(b_tm1, idxs) 
161      shape = self.get_variable_shape(x_tm1) 
162      tm1 = self.get_accumulator(x_tm1, '%s/tm1' % name, shape=[shape[0]]+[1]*(len(shape)-1)) 
163      tm1_ = tf.gather(tm1, idxs) 
164      t = tf.scatter_add(tm1, idxs, tf.ones_like(tm1_)) 
165      t_ = tf.gather(t, idxs) 
166      if beta < 1: 
167        beta_t = tf.convert_to_tensor(beta, name='%s/decay' % name) 
168        beta_t_ = beta_t * (1-beta_t**tm1_) / (1-beta_t**t_) 
169      else: 
170        beta_t_ = tm1_/t_ 
171      b_t = tf.scatter_update(b_tm1, idxs, beta_t_*b_tm1_) 
172      b_t = tf.scatter_add(b_t, idxs, (1-beta_t_)*a_t_) 
173      return b_t, t

176    def _finish(self, caches): 
177      """""" 
178       
179      if self.clip > 0: 
180        S_t = [cache['s_t'] for cache in caches] 
181        S_t, _ = tf.clip_by_global_norm(S_t, self.clip) 
182        for cache, s_t in zip(caches, S_t): 
183          cache['s_t'] = s_t 
184       
185      for cache in caches: 
186        x_tm1 = cache['x_tm1'] 
187        s_t = cache['s_t'] 
188        updates = cache['updates'] 
189        with tf.name_scope('update_' + x_tm1.op.name), tf.device(x_tm1.device): 
190          if 'idxs' in cache: 
191            idxs = cache['idxs'] 
192            x_t = tf.scatter_sub(x_tm1, idxs, s_t) 
193            if self.chi > 0: 
194              x_t_ = tf.gather(x_t, idxs) 
195              x_bar_t, t_x_bar = self._sparse_moving_average(x_tm1, idxs, x_t_, 'x', beta=self.chi) 
196          else: 
197            x_t = tf.assign_sub(x_tm1, s_t) 
198            if self.chi > 0: 
199              x_bar_t, t_x_bar = self._dense_moving_average(x_tm1, x_t, 'x', beta=self.chi) 
200        updates.append(x_t) 
201        if self.chi > 0: 
202          updates.extend([x_bar_t, t_x_bar]) 
203       
204      update_ops = [tf.group(*cache['updates']) for cache in caches] 
205      return tf.group(*update_ops, name='update')

224    def variables_to_restore(self, moving_avg_variables=None): 
225      """""" 
226       
227      name_map = {} 
228      if moving_avg_variables is None: 
229        moving_avg_variables = tf.trainable_variables() 
230        moving_avg_variables += tf.moving_average_variables() 
231      # Remove duplicates 
232      moving_avg_variables = set(moving_avg_variables) 
233      # Collect all the variables with moving average, 
234      for v in moving_avg_variables: 
235        name_map[self.average_name(v)] = v 
236      # Make sure we restore variables without moving average as well. 
237      for v in list(set(tf.all_variables()) - moving_avg_variables): 
238        if v.op.name not in name_map: 
239          name_map[v.op.name] = v 
240      return name_map
